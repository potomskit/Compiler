
class Fibonacci
{
    a = 1;
    b = 1;

    def CalculateNextFibonacciNumber()
    {
        c = self.b;
        self.b = self.a + self.b;
        self.a = c;
        return self.b;
    }
}

def main()
{
    maxValue = 100000;
    fibonacci = Fibonacci();
    fibonacciNumber = 0;
    while(fibonacciNumber < maxValue)
    {
        fibonacciNumber = fibonacci.CalculateNextFibonacciNumber();
    }
    print(fibonacciNumber);
}
